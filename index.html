üß© Basic Usage

Create a window

local UI = Kavo:CreateWindow("My Menu")

Create a tab

local tab = UI:CreateTab("Main")

Create a section

local sec = tab:CreateSection("Features")

Button

sec:AddButton("God Mode", function()
    gg.toast("God Mode Activated!")
end)

Toggle

sec:AddToggle("Speed", false, function(state)
    gg.toast("Speed: " .. tostring(state))
end)

Slider

sec:AddSlider("FOV", 10, 120, 60, function(value)
    gg.toast("FOV = " .. value)
end)

Input

sec:AddInput("Player Name", "User", function(text)
    gg.toast("Name set to: " .. text)
end)

Show UI (Floating Menu)

UI:Show()

üìö Examples

1) Complete basic example

Copy this whole script into a file and run it in GameGuardian (adjust the path to KavoUI.lua if needed).

-- Basic example using local KavoUI.lua
local Kavo = loadfile("/storage/emulated/0/Download/KavoUI.lua")()
Kavo:Welcome("") -- optional: default welcome alert

local win = Kavo:CreateWindow("My Menu")
local mainTab = win:CreateTab("Main")
local general = mainTab:CreateSection("General")

general:AddButton("Say Hello", function()
    gg.toast("Hello from KavoUI!")
end)

general:AddToggle("Fast Mode", false, function(state)
    gg.toast("Fast Mode -> " .. tostring(state))
end)

general:AddSlider("FOV", 10, 120, 60, function(v)
    gg.toast("FOV set to " .. v)
end)

general:AddInput("Player Name", "Player", function(name)
    gg.toast("Player name: " .. name)
end)

win:Show()

2) Dynamic submenu example (pushMenu)

Shows how to open a runtime menu and push nested menus.

-- Dynamic submenu example
local Kavo = loadfile("/storage/emulated/0/Download/KavoUI.lua")()
local win = Kavo:CreateWindow("Dynamic Demo")
local t = win:CreateTab("Tools")
local s = t:CreateSection("Actions")

s:AddButton("Open submenu", function()
    win:pushMenu({
        title = "Runtime Menu",
        options = {
            { text = "Option A", action = function() gg.toast("Option A selected") end },
            { text = "Open deeper", action = function()
                win:pushMenu({
                    title = "Deeper Menu",
                    options = {
                        { text = "Deep 1", action = function() gg.toast("Deep 1") end },
                        { text = "Deep 2", action = function() gg.toast("Deep 2") end },
                    }
                })
            end},
            { text = "Back to UI", action = function() win:backMenu() end },
        }
    })
end)

win:Show()
