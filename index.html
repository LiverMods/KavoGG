Basic Usage

local UI = Kavo:CreateWindow("My Menu")
local tab = UI:CreateTab("Main")
local sec = tab:CreateSection("Features")

Window

local UI = Kavo:CreateWindow("My Menu")

Tab

local tab = UI:CreateTab("Main")

Section

local sec = tab:CreateSection("Features")

Button

sec:AddButton("God Mode", function()
    gg.toast("Activated!")
end)

Toggle

sec:AddToggle("Speed", false, function(state)
    gg.toast("Speed: " .. tostring(state))
end)

Slider

sec:AddSlider("FOV", 10, 120, 60, function(value)
    gg.toast("FOV = " .. value)
end)

Input

sec:AddInput("Player Name", "User", function(text)
    gg.toast("Name: " .. text)
end)

Full Example (Local)

local Kavo = loadfile("/storage/emulated/0/Download/KavoUI.lua")()
Kavo:Welcome("Welcome!")

local win = Kavo:CreateWindow("My Menu")
local tab = win:CreateTab("Main")
local sec  = tab:CreateSection("General")

sec:AddButton("Hello", function()
    gg.toast("Hello from KavoUI!")
end)

sec:AddToggle("Fast Mode", false, function(state)
    gg.toast("Fast Mode: " .. tostring(state))
end)

win:Show()
